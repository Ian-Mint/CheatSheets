%------------ Vector Spaces ---------------
\begin{tikzpicture}
\node [mybox] (box){%
	\begin{minipage}{0.3\textwidth}
	$\forall v_1, v_2, v_3 \in \mathpzc{V}, \; \forall \alpha, \beta \in \mathbb{F}$ \\
	1. $v_1 + v_2 \in \mathpzc{V}$ \\
	2. $\alpha v_1 \in \mathpzc{V} $ \\
	3. $\alpha(v_1 + v_2) = \alpha v_1 + \alpha v_2$ \\
	4. $(\alpha+\beta)v = \alpha v + \beta v$ \\
	5. $\exists 0 \in \mathpzc{V} \mid 0 + v_1 = v_1 + 0 = v_1$ \\
	6. $\exists v_I \in \mathpzc{V} \mid v + v_I = v_I + v = 0 $ \\
	7. $(\alpha \beta)v = \alpha (\beta v)$ \\
	8. $ v_1 + v_2 = v_2 + v_1 $ \\
	9. $1 \in \mathbb{F} \mid 1*v = v$ \\
	10. $(v_1 + v_2) + v_3 = v_1 + (v_2 + v_3) $ \\
	\end{minipage}
};

%------------ Inner Product Spaces ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
	1. $\langle u+v,w\rangle = \langle u,w \rangle + \langle v,w\rangle $ \\
	2. $ \langle \alpha u,v \rangle = \alpha \langle u,v \rangle $ \\
	3. $\langle v,u \rangle = \langle u,v \rangle^* $ \\
	4. $ \langle v,v \rangle \geq 0$ and $ \langle v,v \rangle = 0 \leftrightarrow v = 0 $ \\
	
	Standard Inner Product \\
	$||x|| = \sqrt{\langle x, x \rangle}$ \\
	
	Standard Matrix Inner Product \\
	$\langle X, Y \rangle = Tr(Y^H X)$ \\
	$\langle X, X \rangle  = Tr(X^HX) = \sum_{i, j} |x_{ij}|^2 = ||X||_F^2$
    \end{minipage}
};

	2. $||\alpha x|| = |\alpha| \cdot ||x|| \;\; \forall \alpha \in \mathbb{R},\mathbb{C}$ \\
	3. $||x+y|| \le ||x|| + ||y|| \;\; \forall x,y \in \mathpzc{V}$ \\

	L1 Norm : $|x_1| + |x_2|$
